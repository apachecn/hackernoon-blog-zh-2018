  const comp = styles => {

      let composed = []
      if (styles) {
        composed = [
          ...composed,
          StyleSheet.create({
            styles
          }).styles
        ]
      }

      if (Array.isArray(style)) {
        composed = [...composed, ...style]
      } else if (style) {
        composed = [
          ...composed,
          StyleSheet.create({
            style
          }).style
        ]
      }

      return <Component {...{ style: composed }} {...rest}>{children}</Component>
    }
  }

  return arg => {
    if (typeof arg === 'function') {
      return comp(arg())
    }

    return comp()(arg)
  }
}
